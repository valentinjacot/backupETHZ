006package jdraw.framework;
007
008/**
009 * The class DrawModel represents the model of a drawing, i.e.
010 * all figures stored in a graphic. Every draw view refers to a
011 * model.
012 *
013 * @see DrawView
014 *
015 * @author  Dominik Gruntz &amp; Christoph Denzler
016 * @version 2.5
017 */
018public interface DrawModel {
019
020        /**
021         * Adds a new figure to the draw model.
022         * 
023         * @param f figure to be added to draw model.
024         */
025        void addFigure(Figure f);
026
027        /**
028         * Removes a given figure from the draw model.
029         * 
030         * @param f figure to be removed from draw model.
031         */
032        void removeFigure(Figure f);
033        
034        /**
035         * Remove all figures from the draw model. This method is used when loading a drawing from a file
036         * in order to clear the old drawing before loading the new one.
037         */
038        void removeAllFigures();
039
040        /**
041         * Returns an iterator which can be used to iterate over all figures of a
042         * draw model. The iterator ordering is the order in which elements were
043         * inserted into the model (insertion-order). The order is interpreted by
044         * the view as "back-to-front".
045         * 
046         * @return iterator to iterate over the draw model
047         */
048        Iterable<Figure> getFigures();
049
050        /**
051         * Adds the specified model listener to receive model events from this draw
052         * model.
053         * 
054         * @param listener the draw model listener.
055         * @see DrawModelListener
056         */
057        void addModelChangeListener(DrawModelListener listener);
058
059        /**
060         * Removes the specified model listener so that it no longer receives model
061         * events from this draw model. This method performs no function, nor does
062         * it throw an exception, if the listener specified by the argument was not
063         * previously added to this figure.
064         * 
065         * @param listener the draw model listener.
066         * @see DrawModelListener
067         */
068        void removeModelChangeListener(DrawModelListener listener);
069
070        /**
071         * Returns the draw command handler provided by this model.
072         * 
073         * @see DrawCommandHandler
074         * @return the draw command handler used.
075         */
076        DrawCommandHandler getDrawCommandHandler();
077
078        /**
079         * Sets the index of a given figure. The order of the other figures in the
080         * model remains unchanged.
081         * 
082         * If the figure is moved to a new place, then the model sends a
083         * <code>DRAWING_CHANGED</code> event to all registered model listeners.
084         * 
085         * @param f
086         *            the figure whose index has to be set
087         * @param index
088         *            the position at which the new figure should appear. The other
089         *            figures are moved away.
090         * 
091         * @throws IllegalArgumentException
092         *             if the figure is not contained in the model.
093         * @throws IndexOutOfBoundsException
094         *             if the index is out of range (
095         *             <code>index &lt; 0 || index &ge; size()</code>) where size() is
096         *             the number of figures contained in the model.
097         */
098        void setFigureIndex(Figure f, int index) throws IllegalArgumentException, IndexOutOfBoundsException;
099
100}
